const DraggableNode = ({ id, x, y, icon: Icon, label, color, onDrag }) => {
  const [isDragging, setIsDragging] = useState(false);
  const [position, setPosition] = useState({ x, y });
  const offset = useRef({ x: 0, y: 0 });

  const handleMouseDown = (e) => {
    setIsDragging(true);
    const rect = e.currentTarget.getBoundingClientRect();
    offset.current = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
  };

  useEffect(() => {
    const handleMouseMove = (e) => {
      if (!isDragging) return;
      
      // Calculate new position relative to the container (roughly)
      // In a real app we'd use the container ref, simplifying here for the prototype
      const newX = e.clientX - 320 - offset.current.x; // 320 is approx sidebar width
      const newY = e.clientY - 80 - offset.current.y;   // 80 is approx header height
      
      const newPos = { x: newX, y: newY };
      setPosition(newPos);
      onDrag(id, newPos);
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, id, onDrag]);

  const borderColors = {
    blue: "border-blue-500 shadow-[0_0_15px_rgba(59,130,246,0.3)]",
    indigo: "border-indigo-500 shadow-[0_0_15px_rgba(99,102,241,0.3)]",
    cyan: "border-cyan-500 shadow-[0_0_15px_rgba(6,182,212,0.3)]",
    emerald: "border-emerald-500 shadow-[0_0_15px_rgba(16,185,129,0.3)]"
  };

  return (
    <div 
      onMouseDown={handleMouseDown}
      style={{ left: position.x, top: position.y }}
      className={`absolute w-32 h-32 rounded-full bg-slate-900 border-2 ${borderColors[color]} flex flex-col items-center justify-center gap-2 cursor-grab active:cursor-grabbing z-10 select-none`}
    >
      <Icon className="text-white" size={24} />
      <span className="text-xs font-bold text-white">{label}</span>
      <div className="w-2 h-2 rounded-full bg-white absolute -right-1 top-1/2 transform -translate-y-1/2"></div>
      <div className="w-2 h-2 rounded-full bg-white absolute -left-1 top-1/2 transform -translate-y-1/2"></div>
    </div>
  );
};

const WorkflowView = () => {
  // Initial positions matching the screenshot curve
  const [nodes, setNodes] = useState([
    { id: 1, label: "Start", icon: Zap, x: 100, y: 300, color: "emerald" },
    { id: 2, label: "Plan #1", icon: FileText, x: 300, y: 150, color: "blue" },
    { id: 3, label: "Video #2", icon: Video, x: 600, y: 150, color: "indigo" },
    { id: 4, label: "Publish #3", icon: Send, x: 800, y: 400, color: "cyan" }
  ]);

  const handleDrag = (id, newPos) => {
    setNodes(nodes.map(n => n.id === id ? { ...n, ...newPos } : n));
  };

  // Simple bezier curve generator between nodes
  const generatePath = () => {
    let path = "";
    for (let i = 0; i < nodes.length - 1; i++) {
      const current = nodes[i];
      const next = nodes[i+1];
      // Offset to connect from edge of circle not center (approx)
      const startX = current.x + 64; 
      const startY = current.y + 64;
      const endX = next.x + 64;
      const endY = next.y + 64;
      
      const cp1X = startX + (endX - startX) / 2;
      const cp1Y = startY;
      const cp2X = startX + (endX - startX) / 2;
      const cp2Y = endY;

      if (i === 0) path += `M ${startX} ${startY}`;
      path += ` C ${cp1X} ${cp1Y}, ${cp2X} ${cp2Y}, ${endX} ${endY}`;
    }
    return path;
  };

  return (
    <div className="h-full bg-slate-950 relative overflow-hidden cursor-crosshair">
      <div className="absolute top-6 right-6 z-20 flex gap-3">
        <Button variant="secondary">Save Draft</Button>
        <Button>Create Agent</Button>
      </div>

      {/* SVG Layer for Connections */}
      <svg className="absolute inset-0 w-full h-full pointer-events-none z-0">
        <path 
          d={generatePath()} 
          stroke="#3b82f6" 
          strokeWidth="4" 
          fill="none" 
          strokeLinecap="round"
          className="opacity-50"
        />
        {/* Animated Dash */}
        <path 
          d={generatePath()} 
          stroke="#60a5fa" 
          strokeWidth="4" 
          fill="none" 
          strokeLinecap="round"
          strokeDasharray="10,10"
          className="animate-[dash_20s_linear_infinite]"
        />
      </svg>

      {/* Nodes */}
      {nodes.map(node => (
        <DraggableNode 
          key={node.id} 
          {...node} 
          onDrag={handleDrag} 
        />
      ))}

      <div className="absolute bottom-6 left-1/2 transform -translate-x-1/2 flex gap-2 p-2 bg-slate-900 border border-slate-800 rounded-lg">
        <button className="p-2 hover:bg-slate-800 rounded text-slate-400"><Search size={16} /></button>
        <span className="px-2 py-1 text-xs text-slate-400 flex items-center">100%</span>
        <button className="p-2 hover:bg-slate-800 rounded text-slate-400"><Workflow size={16} /></button>
      </div>
    </div>
  );
};

// --- Custom Draggable Workflow Node Logic ---
const DraggableNode = ({ id, x, y, icon: Icon, label, color, onDrag }: any) => {
  const [isDragging, setIsDragging] = useState(false);
  const [position, setPosition] = useState({ x, y });
  const offset = useRef({ x: 0, y: 0 });

  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
    const rect = e.currentTarget.getBoundingClientRect();
    offset.current = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
  };

  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => {
      if (!isDragging) return;
      
      // Calculate new position relative to the container (roughly)
      // In a real app we'd use the container ref, simplifying here for the prototype
      const newX = e.clientX - 256 - offset.current.x; // 256 is sidebar width
      const newY = e.clientY - offset.current.y;
      
      const newPos = { x: newX, y: newY };
      setPosition(newPos);
      onDrag(id, newPos);
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, id, onDrag]);

  const borderColors: any = {
    blue: "border-blue-500 shadow-[0_0_20px_rgba(59,130,246,0.2)]",
    indigo: "border-indigo-500 shadow-[0_0_20px_rgba(99,102,241,0.2)]",
    cyan: "border-cyan-500 shadow-[0_0_20px_rgba(6,182,212,0.2)]",
    emerald: "border-emerald-500 shadow-[0_0_20px_rgba(16,185,129,0.2)]"
  };

  return (
    <div 
      onMouseDown={handleMouseDown}
      style={{ left: position.x, top: position.y }}
      className={`absolute w-24 h-24 rounded-full bg-slate-950 border-2 ${borderColors[color]} flex flex-col items-center justify-center gap-2 cursor-grab active:cursor-grabbing z-10 select-none group transition-shadow hover:shadow-[0_0_30px_rgba(59,130,246,0.4)]`}
    >
      <Icon className="text-white" size={24} />
      <span className="text-[10px] font-bold text-white uppercase tracking-wide mt-1 bg-slate-900 px-2 py-0.5 rounded-full border border-slate-800">{label}</span>
      
      {/* Node Connection Points */}
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -right-1.5 top-1/2 transform -translate-y-1/2"></div>
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -left-1.5 top-1/2 transform -translate-y-1/2"></div>
    </div>
  );
};

const WorkflowView = () => {
  // Initial positions matching the screenshot curve roughly
  const [nodes, setNodes] = useState([
    { id: 1, label: "Start", icon: Zap, x: 100, y: 300, color: "emerald" },
    { id: 2, label: "Plan #1", icon: FileText, x: 300, y: 150, color: "blue" },
    { id: 3, label: "Video #2", icon: Video, x: 600, y: 150, color: "indigo" },
    { id: 4, label: "Publish #3", icon: Send, x: 800, y: 400, color: "cyan" }
  ]);

  const handleDrag = (id: number, newPos: {x: number, y: number}) => {
    setNodes(nodes.map(n => n.id === id ? { ...n, ...newPos } : n));
  };

  // Simple bezier curve generator between nodes
  const generatePath = () => {
    let path = "";
    for (let i = 0; i < nodes.length - 1; i++) {
      const current = nodes[i];
      const next = nodes[i+1];
      // Offset to connect from edge of circle not center
      const startX = current.x + 96; // width
      const startY = current.y + 48; // half height
      const endX = next.x;
      const endY = next.y + 48;
      
      const cp1X = startX + (endX - startX) / 2;
      const cp1Y = startY;
      const cp2X = startX + (endX - startX) / 2;
      const cp2Y = endY;

      if (i === 0) path += `M ${startX} ${startY}`;
      // Add a MoveTo for disconnected paths if we supported branching, 
      // but for this chain we just line to. 
      // Actually SVG paths need to be continuous or use multiple d attributes.
      // For a single chain:
      if (i > 0) path += ` L ${startX} ${startY}`; // Draw line from prev end to current start? 
      // Wait, standard node editors draw independent lines.
      // Let's return an array of paths instead to handle moving nodes correctly.
    }
    return path;
  };

  return (
    <div className="h-full bg-slate-950 relative overflow-hidden cursor-crosshair">
      {/* Background Grid */}
      <div className="absolute inset-0 bg-[linear-gradient(to_right,#1e293b_1px,transparent_1px),linear-gradient(to_bottom,#1e293b_1px,transparent_1px)] bg-[size:40px_40px] opacity-20 pointer-events-none" />

      <div className="absolute top-6 right-6 z-20 flex gap-3">
        <Button variant="secondary" icon={FileText}>Save Draft</Button>
        <Button icon={Plus}>Create Agent</Button>
      </div>

      {/* SVG Layer for Connections */}
      <svg className="absolute inset-0 w-full h-full pointer-events-none z-0">
        <defs>
          <linearGradient id="gradient" x1="0%" y1="0%" x2="100%" y2="0%">
            <stop offset="0%" stopColor="#10b981" />
            <stop offset="50%" stopColor="#3b82f6" />
            <stop offset="100%" stopColor="#06b6d4" />
          </linearGradient>
        </defs>
        {nodes.map((node, i) => {
          if (i === nodes.length - 1) return null;
          const next = nodes[i+1];
          const startX = node.x + 96; // Right edge
          const startY = node.y + 48; // Center Y
          const endX = next.x;        // Left edge
          const endY = next.y + 48;   // Center Y
          
          const cp1X = startX + (endX - startX) / 2;
          const cp2X = startX + (endX - startX) / 2;

          const d = `M ${startX} ${startY} C ${cp1X} ${startY}, ${cp2X} ${endY}, ${endX} ${endY}`;

          return (
            <g key={i}>
              {/* Glow Effect */}
              <path d={d} stroke="rgba(59, 130, 246, 0.3)" strokeWidth="8" fill="none" />
              {/* Main Line */}
              <path d={d} stroke="url(#gradient)" strokeWidth="3" fill="none" strokeLinecap="round" />
              {/* Animated Dash */}
              <path 
                d={d} 
                stroke="white" 
                strokeWidth="2" 
                fill="none" 
                strokeDasharray="4,8"
                className="opacity-50 animate-[dash_1s_linear_infinite]"
              />
            </g>
          )
        })}
      </svg>

      {/* Nodes */}
      {nodes.map(node => (
        <DraggableNode 
          key={node.id} 
          {...node} 
          onDrag={handleDrag} 
        />
      ))}

      {/* Zoom Controls */}
      <div className="absolute bottom-6 left-1/2 transform -translate-x-1/2 flex gap-1 p-1 bg-slate-900/90 border border-slate-800 rounded-lg backdrop-blur shadow-xl">
        <button className="p-2 hover:bg-slate-800 rounded text-slate-400 transition-colors"><ZoomOut size={16} /></button>
        <span className="px-3 py-2 text-xs text-slate-300 font-mono flex items-center min-w-[50px] justify-center bg-slate-950 rounded">100%</span>
        <button className="p-2 hover:bg-slate-800 rounded text-slate-400 transition-colors"><ZoomIn size={16} /></button>
        <div className="w-px h-6 bg-slate-800 mx-1 self-center"></div>
        <button className="p-2 hover:bg-slate-800 rounded text-slate-400 transition-colors"><Maximize size={16} /></button>
      </div>
    </div>
  );
};

// --- Custom Draggable Workflow Node Logic ---
const DraggableNode = ({ id, x, y, icon: Icon, label, color, onDrag }: any) => {
  const [isDragging, setIsDragging] = useState(false);
  const [position, setPosition] = useState({ x, y });
  const offset = useRef({ x: 0, y: 0 });

  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
    const rect = e.currentTarget.getBoundingClientRect();
    offset.current = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
  };

  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => {
      if (!isDragging) return;
      
      // Calculate new position relative to the container (roughly)
      // In a real app we'd use the container ref, simplifying here for the prototype
      const newX = e.clientX - 256 - offset.current.x; // 256 is sidebar width
      const newY = e.clientY - offset.current.y;
      
      const newPos = { x: newX, y: newY };
      setPosition(newPos);
      onDrag(id, newPos);
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, id, onDrag]);

  const borderColors: any = {
    blue: "border-blue-500 shadow-[0_0_20px_rgba(59,130,246,0.2)]",
    indigo: "border-indigo-500 shadow-[0_0_20px_rgba(99,102,241,0.2)]",
    cyan: "border-cyan-500 shadow-[0_0_20px_rgba(6,182,212,0.2)]",
    emerald: "border-emerald-500 shadow-[0_0_20px_rgba(16,185,129,0.2)]"
  };

  return (
    <div 
      onMouseDown={handleMouseDown}
      style={{ left: position.x, top: position.y }}
      className={`absolute w-24 h-24 rounded-full bg-slate-950 border-2 ${borderColors[color]} flex flex-col items-center justify-center gap-2 cursor-grab active:cursor-grabbing z-10 select-none group transition-shadow hover:shadow-[0_0_30px_rgba(59,130,246,0.4)]`}
    >
      <Icon className="text-white" size={24} />
      <span className="text-[10px] font-bold text-white uppercase tracking-wide mt-1 bg-slate-900 px-2 py-0.5 rounded-full border border-slate-800">{label}</span>
      
      {/* Node Connection Points */}
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -right-1.5 top-1/2 transform -translate-y-1/2"></div>
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -left-1.5 top-1/2 transform -translate-y-1/2"></div>
    </div>
  );
};

// --- Custom Draggable Workflow Node Logic ---
const DraggableNode = ({ id, x, y, icon: Icon, label, color, onDrag }: any) => {
  const [isDragging, setIsDragging] = useState(false);
  const [position, setPosition] = useState({ x, y });
  const offset = useRef({ x: 0, y: 0 });

  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
    const rect = e.currentTarget.getBoundingClientRect();
    offset.current = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
  };

  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => {
      if (!isDragging) return;
      
      // Calculate new position relative to the container (roughly)
      // In a real app we'd use the container ref, simplifying here for the prototype
      const newX = e.clientX - 256 - offset.current.x; // 256 is sidebar width
      const newY = e.clientY - offset.current.y;
      
      const newPos = { x: newX, y: newY };
      setPosition(newPos);
      onDrag(id, newPos);
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, id, onDrag]);

  const borderColors: any = {
    blue: "border-blue-500 shadow-[0_0_20px_rgba(59,130,246,0.2)]",
    indigo: "border-indigo-500 shadow-[0_0_20px_rgba(99,102,241,0.2)]",
    cyan: "border-cyan-500 shadow-[0_0_20px_rgba(6,182,212,0.2)]",
    emerald: "border-emerald-500 shadow-[0_0_20px_rgba(16,185,129,0.2)]"
  };

  return (
    <div 
      onMouseDown={handleMouseDown}
      style={{ left: position.x, top: position.y }}
      className={`absolute w-24 h-24 rounded-full bg-slate-950 border-2 ${borderColors[color]} flex flex-col items-center justify-center gap-2 cursor-grab active:cursor-grabbing z-10 select-none group transition-shadow hover:shadow-[0_0_30px_rgba(59,130,246,0.4)]`}
    >
      <Icon className="text-white" size={24} />
      <span className="text-[10px] font-bold text-white uppercase tracking-wide mt-1 bg-slate-900 px-2 py-0.5 rounded-full border border-slate-800">{label}</span>
      
      {/* Node Connection Points */}
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -right-1.5 top-1/2 transform -translate-y-1/2"></div>
      <div className="w-3 h-3 rounded-full bg-white border-2 border-slate-900 absolute -left-1.5 top-1/2 transform -translate-y-1/2"></div>
    </div>
  );
};

// --- Custom Draggable Workflow Node Logic ---
const DraggableNode = ({ id, x, y, icon: Icon, label, color, onDrag }: any) => {
  const [isDragging, setIsDragging] = useState(false);
  const [position, setPosition] = useState({ x, y });
  const offset = useRef({ x: 0, y: 0 });

  const handleMouseDown = (e: React.MouseEvent) => {
    setIsDragging(true);
    const rect = e.currentTarget.getBoundingClientRect();
    offset.current = {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top
    };
  };

  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => {
      if (!isDragging) return;
      
      // Calculate new position relative to the container (roughly)
      // In a real app we'd use the container ref, simplifying here for the prototype
      const newX = e.clientX - 256 - offset.current.x; // 256 is sidebar width
      const newY = e.clientY - offset.current.y;
      
      const newPos = { x: newX, y: newY };
      setPosition(newPos);
      onDrag(id, newPos);
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

import { nanoid } from 'nanoid';

export type AgentStatus = 'idle' | 'working' | 'completed' | 'failed';

export interface AgentTask {
  id: string;
  type: 'research' | 'script' | 'visual' | 'publish';
  status: AgentStatus;
  output?: string;
  progress: number;
}

export interface ProjectState {
  id: string;
  name: string;
  status: 'planning' | 'researching' | 'scripting' | 'generating' | 'publishing' | 'completed';
  tasks: AgentTask[];
  currentStep: number;
}

// Mock responses for the simulation
const MOCK_RESEARCH = [
  "Analyzing trending topics in AI...",
  "Found 3 viral hooks on Twitter...",
  "Cross-referencing with YouTube trends...",
  "Research complete. Identified key narrative angle."
];

const MOCK_SCRIPT = [
  "Drafting intro hook...",
  "Structuring main body points...",
  "Refining call-to-action...",
  "Script generated. Tone: Professional & Engaging."
];

const MOCK_VISUALS = [
  "Generating scene descriptions...",
  "Creating image prompts for DALL-E 3...",
  "Rendering scene 1 of 4...",
  "Visual assets ready for assembly."
];

export class AgentSimulator {
  private listeners: ((state: ProjectState) => void)[] = [];
  private state: ProjectState;

  constructor(initialName: string) {
    this.state = {
      id: nanoid(),
      name: initialName,
      status: 'planning',
      currentStep: 0,
      tasks: [
        { id: '1', type: 'research', status: 'idle', progress: 0 },
        { id: '2', type: 'script', status: 'idle', progress: 0 },
        { id: '3', type: 'visual', status: 'idle', progress: 0 },
        { id: '4', type: 'publish', status: 'idle', progress: 0 }
      ]
    };
  }

  subscribe(listener: (state: ProjectState) => void) {
    this.listeners.push(listener);
    return () => {
      this.listeners = this.listeners.filter(l => l !== listener);
    };
  }

  private notify() {
    this.listeners.forEach(l => l({ ...this.state }));
  }

  async start() {
    // Step 1: Research
    this.state.status = 'researching';
    this.state.tasks[0].status = 'working';
    this.notify();
    
    await this.simulateTask(0, MOCK_RESEARCH);
    this.state.tasks[0].status = 'completed';
    this.state.currentStep = 1;
    this.notify();

    // Step 2: Script
    this.state.status = 'scripting';
    this.state.tasks[1].status = 'working';
    this.notify();

    await this.simulateTask(1, MOCK_SCRIPT);
    this.state.tasks[1].status = 'completed';
    this.state.currentStep = 2;
    this.notify();

    // Step 3: Visuals
    this.state.status = 'generating';
    this.state.tasks[2].status = 'working';
    this.notify();

    await this.simulateTask(2, MOCK_VISUALS);
    this.state.tasks[2].status = 'completed';
    this.state.currentStep = 3;
    this.notify();

    // Step 4: Publish
    this.state.status = 'publishing';
    this.state.tasks[3].status = 'working';
    this.notify();

    await this.simulateTask(3, ["Connecting to YouTube API...", "Uploading video...", "Published successfully!"]);
    this.state.tasks[3].status = 'completed';
    this.state.status = 'completed';
    this.state.currentStep = 4;
    this.notify();
  }

  private async simulateTask(taskIndex: number, logs: string[]) {
    for (let i = 0; i < logs.length; i++) {
      await new Promise(resolve => setTimeout(resolve, 1500)); 
      this.state.tasks[taskIndex].output = logs[i];
      this.state.tasks[taskIndex].progress = ((i + 1) / logs.length) * 100;
      this.notify();
    }
  }
}

import { motion } from "framer-motion";
import { Bot, FileText, Send, Video } from "lucide-react";

interface StepProps {
  icon: React.ElementType;
  label: string;
  status: 'pending' | 'active' | 'completed';
  delay?: number;
}

function FlowStep({ icon: Icon, label, status, delay = 0 }: StepProps) {
  return (
    <div className="flex flex-col items-center gap-3 relative z-10">
      <motion.div
        initial={{ scale: 0.8, opacity: 0.5 }}
        animate={{ 
          scale: status === 'active' ? 1.1 : 1, 
          opacity: status === 'pending' ? 0.5 : 1,
          borderColor: status === 'active' ? '#3b82f6' : status === 'completed' ? '#3b82f6' : '#334155',
          boxShadow: status === 'active' ? '0 0 20px rgba(59,130,246,0.5)' : 'none'
        }}
        transition={{ duration: 0.5, delay }}
        className={`h-16 w-16 rounded-full bg-[#0f1016] border-2 flex items-center justify-center
          ${status === 'active' ? 'border-blue-500' : 
            status === 'completed' ? 'border-blue-500' : 'border-slate-700'}`}
      >
        <Icon className={`h-6 w-6 ${status === 'pending' ? 'text-slate-500' : 'text-white'}`} />
      </motion.div>
      <span className={`text-sm font-medium ${status === 'pending' ? 'text-slate-500' : 'text-white'}`}>
        {label}
      </span>
    </div>
  );
}

export function FlowCanvas({ activeStep }: { activeStep: number }) {
  return (
    <div className="flex-1 bg-[#05050a] relative overflow-hidden flex items-center justify-center">
      {/* Background Grid */}
      <div className="absolute inset-0 bg-[linear-gradient(to_right,#ffffff05_1px,transparent_1px),linear-gradient(to_bottom,#ffffff05_1px,transparent_1px)] bg-[size:40px_40px]"></div>
      
      {/* Connecting Line (SVG) */}
      <svg className="absolute inset-0 w-full h-full pointer-events-none">
        {/* Background Dashed Line */}
        <motion.path
          d="M 300 400 C 400 400, 400 300, 500 300 C 600 300, 600 300, 700 300 C 800 300, 800 500, 900 500"
          fill="none"
          stroke="#1e293b"
          strokeWidth="2"
          strokeDasharray="5 5"
        />

4. Agent Logic (src/lib/agents.ts)
Simulates the "Vibe Coding" agent workflow.

TypeScript
import { nanoid } from 'nanoid';

export type AgentStatus = 'idle' | 'working' | 'completed' | 'failed';

export interface AgentTask {
  id: string;
  type: 'research' | 'script' | 'visual' | 'publish';
  status: AgentStatus;
  output?: string;
  progress: number;
}

export interface ProjectState {
  id: string;
  name: string;
  status: 'planning' | 'researching' | 'scripting' | 'generating' | 'publishing' | 'completed';
  tasks: AgentTask[];
  currentStep: number;
}

// Mock responses for the simulation
const MOCK_RESEARCH = [
  "Analyzing trending topics in AI...",
  "Found 3 viral hooks on Twitter...",
  "Cross-referencing with YouTube trends...",
  "Research complete. Identified key narrative angle."
];

const MOCK_SCRIPT = [
  "Drafting intro hook...",
  "Structuring main body points...",
  "Refining call-to-action...",
  "Script generated. Tone: Professional & Engaging."
];

const MOCK_VISUALS = [
  "Generating scene descriptions...",
  "Creating image prompts for DALL-E 3...",
  "Rendering scene 1 of 4...",
  "Visual assets ready for assembly."
];

export class AgentSimulator {
  private listeners: ((state: ProjectState) => void)[] = [];
  private state: ProjectState;

  constructor(initialName: string) {
    this.state = {
      id: nanoid(),
      name: initialName,
      status: 'planning',
      currentStep: 0,
      tasks: [
        { id: '1', type: 'research', status: 'idle', progress: 0 },
        { id: '2', type: 'script', status: 'idle', progress: 0 },
        { id: '3', type: 'visual', status: 'idle', progress: 0 },
        { id: '4', type: 'publish', status: 'idle', progress: 0 }
      ]
    };
  }

  subscribe(listener: (state: ProjectState) => void) {
    this.listeners.push(listener);
    return () => {
      this.listeners = this.listeners.filter(l => l !== listener);
    };
  }

  private notify() {
    this.listeners.forEach(l => l({ ...this.state }));
  }

  async start() {
    // Step 1: Research
    this.state.status = 'researching';
    this.state.tasks[0].status = 'working';
    this.notify();
    
    await this.simulateTask(0, MOCK_RESEARCH);
    this.state.tasks[0].status = 'completed';
    this.state.currentStep = 1;
    this.notify();

    // Step 2: Script
    this.state.status = 'scripting';
    this.state.tasks[1].status = 'working';
    this.notify();

    await this.simulateTask(1, MOCK_SCRIPT);
    this.state.tasks[1].status = 'completed';
    this.state.currentStep = 2;
    this.notify();

    // Step 3: Visuals
    this.state.status = 'generating';
    this.state.tasks[2].status = 'working';
    this.notify();

    await this.simulateTask(2, MOCK_VISUALS);
    this.state.tasks[2].status = 'completed';
    this.state.currentStep = 3;
    this.notify();

    // Step 4: Publish
    this.state.status = 'publishing';
    this.state.tasks[3].status = 'working';
    this.notify();

    await this.simulateTask(3, ["Connecting to YouTube API...", "Uploading video...", "Published successfully!"]);
    this.state.tasks[3].status = 'completed';
    this.state.status = 'completed';
    this.state.currentStep = 4;
    this.notify();
  }

  private async simulateTask(taskIndex: number, logs: string[]) {
    for (let i = 0; i < logs.length; i++) {
      await new Promise(resolve => setTimeout(resolve, 1500)); 
      this.state.tasks[taskIndex].output = logs[i];
      this.state.tasks[taskIndex].progress = ((i + 1) / logs.length) * 100;
      this.notify();
    }
  }
}
